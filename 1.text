<script>
(() => {
  const track=document.getElementById('hiDailyTrack'); if(!track) return;
  const slides=[...track.querySelectorAll('.hiM-slide')];

  const setActive=i=>slides.forEach((s,k)=>s.toggleAttribute('data-active',k===i));
  const go=i=>{ i=Math.max(0,Math.min(i,slides.length-1)); track.scrollTo({left:slides[i].offsetLeft,behavior:'smooth'}); setActive(i); };

  // شروع از راست‌ترین اسلاید (بعد از load و بعد از لود تصاویر لِیزی)
  function snapRight(){
    const last=slides.at(-1);
    if(!last) return;
    track.scrollLeft = last.offsetLeft;
    setActive(slides.length-1);
  }
  window.addEventListener('load', snapRight);
  slides.forEach(sl=>{ const img=sl.querySelector('img'); if(img && !img.complete){ img.addEventListener('load', snapRight, {once:true}); }});
  window.addEventListener('resize', snapRight);

  // کنترل‌ها
  const prev=track.parentElement.querySelector('.hiM-btn.prev');
  const next=track.parentElement.querySelector('.hiM-btn.next');
  const cur=()=>slides.findIndex(s=>s.hasAttribute('data-active'));
  prev&&prev.addEventListener('click',()=>go(cur()-1));
  next&&next.addEventListener('click',()=>go(cur()+1));

  // فقط با کلیک روی ! باز/بسته شود (Event Delegation)
  track.addEventListener('click', (e)=>{
    const info = e.target.closest('.hiM-infoIcon');
    if(info){
      const slide = info.closest('.hiM-slide');
      const syn = slide.querySelector('.hiM-syn');
      const open=!syn.classList.contains('open');
      syn.classList.toggle('open', open);
      info.setAttribute('aria-expanded', open ? 'true' : 'false');
      e.stopPropagation(); e.preventDefault();
      return;
    }
    const close = e.target.closest('.hiM-synClose');
    if(close){
      const slide=close.closest('.hiM-slide');
      slide.querySelector('.hiM-syn').classList.remove('open');
      slide.querySelector('.hiM-infoIcon')?.setAttribute('aria-expanded','false');
      e.stopPropagation(); e.preventDefault();
    }
  }, true); // capture=true تا روی همه لایه‌ها بگیرد
})();
</script>
